Custom Nodes
Core
ℹ️ What is a Node?
A node is the fundamental unit of computation in ROS 2. Each node runs as an independent process, communicates via topics/services/actions, and can publish/subscribe to data. Nodes are typically specialized (one node per sensor, one for processing, etc.).
Minimal Python Node
python
my_node.py
import rclpy
from rclpy.node import Node

class MyNode(Node):
    def __init__(self):
        super().__init__('my_node')          # node name in ROS graph
        self.get_logger().info('Node started!')
        self.get_logger().warn('This is a warning')
        self.get_logger().error('This is an error')
        self.get_logger().debug('Debug message (hidden by default)')

        # Create a timer (calls callback every 1.0 sec)
        self.timer = self.create_timer(1.0, self.timer_callback)
        self.counter = 0

    def timer_callback(self):
        self.counter += 1
        self.get_logger().info(f'Count: {self.counter}')

def main(args=None):
    rclpy.init(args=args)          # initialize rclpy
    node = MyNode()                # instantiate node
    rclpy.spin(node)               # block here, process callbacks
    node.destroy_node()            # cleanup
    rclpy.shutdown()               # shutdown rclpy

if __name__ == '__main__':
    main()

Topics — Publisher & Subscriber
Pub/Sub
ℹ️ Topics
Topics are named buses over which nodes exchange messages. Publishers send messages; subscribers receive them. Completely decoupled — publisher doesn't know who is listening.
Publisher + Subscriber in One Node (Python)
python
pub_sub_node.py
import rclpy
from rclpy.node import Node
from std_msgs.msg import String, Int32, Float64
from geometry_msgs.msg import Twist

class PubSubNode(Node):
    def __init__(self):
        super().__init__('pub_sub_node')

        # --- PUBLISHER ---
        # create_publisher(MsgType, topic_name, queue_size)
        self.pub = self.create_publisher(String, '/chatter', 10)

        # --- SUBSCRIBER ---
        # create_subscription(MsgType, topic_name, callback, queue_size)
        self.sub = self.create_subscription(
            String, '/chatter', self.listener_callback, 10)

        # --- TIMER ---
        self.timer = self.create_timer(0.5, self.timer_callback)
        self.count = 0

    def timer_callback(self):
        msg = String()
        msg.data = f'Hello World: {self.count}'
        self.pub.publish(msg)
        self.count += 1

    def listener_callback(self, msg: String):
        self.get_logger().info(f'Received: {msg.data}')

def main(args=None):
    rclpy.init(args=args)
    node = PubSubNode()
    rclpy.spin(node)
    rclpy.shutdown()

Services
Request/Response
ℹ️ Services
Services implement a request/response communication pattern. A client sends a request and waits synchronously for a response. Good for discrete one-time actions (get sensor value, enable motor, etc.).
Service Server (Python)
python
add_server.py
import rclpy
from rclpy.node import Node
from example_interfaces.srv import AddTwoInts  # built-in example srv

class AddServer(Node):
    def __init__(self):
        super().__init__('add_server')
        # create_service(SrvType, service_name, callback)
        self.srv = self.create_service(
            AddTwoInts, 'add_two_ints', self.add_callback)
        self.get_logger().info('Service ready: add_two_ints')

    def add_callback(self, request, response):
        # request and response are generated from the .srv file
        response.sum = request.a + request.b
        self.get_logger().info(f'{request.a} + {request.b} = {response.sum}')
        return response  # must return response object

def main(args=None):
    rclpy.init(args=args)
    node = AddServer()
    rclpy.spin(node)
    rclpy.shutdown()
Service Client (Python)
python
add_client.py
import rclpy
from rclpy.node import Node
from example_interfaces.srv import AddTwoInts

class AddClient(Node):
    def __init__(self):
        super().__init__('add_client')
        # create_client(SrvType, service_name)
        self.client = self.create_client(AddTwoInts, 'add_two_ints')

    def send_request(self, a, b):
        # Wait for service to be available
        while not self.client.wait_for_service(timeout_sec=1.0):
            self.get_logger().warn('Waiting for service...')

        req = AddTwoInts.Request()
        req.a = a
        req.b = b

        # Async call — returns a Future
        future = self.client.call_async(req)
        rclpy.spin_until_future_complete(self, future)

        if future.result() is not None:
            self.get_logger().info(f'Result: {future.result().sum}')
            return future.result().sum
        else:
            self.get_logger().error('Service call failed!')
            return None

def main(args=None):
    rclpy.init(args=args)
    client = AddClient()
    result = client.send_request(5, 3)
    print(f'5 + 3 = {result}')
    client.destroy_node()
    rclpy.shutdown()

#include "rclcpp/rclcpp.hpp"
#include "example_interfaces/srv/add_two_ints.hpp"

class AddServer : public rclcpp::Node {
public:
    AddServer() : Node("add_server") {
        srv_ = create_service<example_interfaces::srv::AddTwoInts>(

                         example_interfaces::srv::AddTwoInts::Response::SharedPtr res) {
                res->sum = req->a + req->b;
                RCLCPP_INFO(get_logger(), "%ld + %ld = %ld", req->a, req->b, res->sum);
            });
    }
private:
    rclcpp::Service<example_interfaces::srv::AddTwoInts>::SharedPtr srv_;
};
cpp
#include "rclcpp/rclcpp.hpp"
#include "example_interfaces/srv/add_two_ints.hpp"

int main(int argc, char* argv[]) {
    rclcpp::init(argc, argv);
    auto node = std::make_shared<rclcpp::Node>("add_client");
    auto client = node->create_client<example_interfaces::srv::AddTwoInts>("add_two_ints");

    while (!client->wait_for_service(std::chrono::seconds(1)))
        RCLCPP_WARN(node->get_logger(), "Waiting for service...");

    auto request = std::make_shared<example_interfaces::srv::AddTwoInts::Request>();
    request->a = 5; request->b = 3;

    auto future = client->async_send_request(request);
    if (rclcpp::spin_until_future_complete(node, future) == rclcpp::FutureReturnCode::SUCCESS)
        RCLCPP_INFO(node->get_logger(), "Result: %ld", future.get()->sum);

    rclcpp::shutdown();
    return 0;
}

Actions
Long Tasks
ℹ️ Actions
Actions are for long-running tasks with feedback. They combine a goal (what to do), feedback (progress), and result (final outcome). Can be cancelled mid-execution. Built on top of topics and services.
Action Server (Python)
python
fib_server.py
import time, rclpy
from rclpy.node import Node
from rclpy.action import ActionServer, CancelResponse, GoalResponse
from example_interfaces.action import Fibonacci

class FibonacciActionServer(Node):
    def __init__(self):
        super().__init__('fibonacci_server')
        self._server = ActionServer(
            self,
            Fibonacci,
            'fibonacci',
            execute_callback=self.execute_cb,
            goal_callback=self.goal_cb,        # optional: accept/reject goal
            cancel_callback=self.cancel_cb,    # optional: accept/reject cancel
        )
        self.get_logger().info('Action server ready')

    def goal_cb(self, goal_request):
        self.get_logger().info(f'Received goal: order={goal_request.order}')
        return GoalResponse.ACCEPT  # or REJECT

    def cancel_cb(self, goal_handle):
        self.get_logger().info('Received cancel request')
        return CancelResponse.ACCEPT

    def execute_cb(self, goal_handle):
        self.get_logger().info('Executing goal...')
        feedback_msg = Fibonacci.Feedback()
        feedback_msg.partial_sequence = [0, 1]

        for i in range(1, goal_handle.request.order):
            if goal_handle.is_cancel_requested:
                goal_handle.canceled()
                return Fibonacci.Result()

            feedback_msg.partial_sequence.append(
                feedback_msg.partial_sequence[-1] + feedback_msg.partial_sequence[-2])

            goal_handle.publish_feedback(feedback_msg)  # send progress
            time.sleep(0.5)

        goal_handle.succeed()
        result = Fibonacci.Result()
        result.sequence = feedback_msg.partial_sequence
        return result

def main(args=None):
    rclpy.init(args=args)
    node = FibonacciActionServer()
    rclpy.spin(node)
    rclpy.shutdown()
Action Client (Python)
python
fib_client.py
import rclpy
from rclpy.node import Node
from rclpy.action import ActionClient
from example_interfaces.action import Fibonacci

class FibonacciClient(Node):
    def __init__(self):
        super().__init__('fibonacci_client')
        self._client = ActionClient(self, Fibonacci, 'fibonacci')

    def send_goal(self, order):
        self._client.wait_for_server()
        goal = Fibonacci.Goal()
        goal.order = order

        self.get_logger().info(f'Sending goal: order={order}')
        self._send_goal_future = self._client.send_goal_async(
            goal,
            feedback_callback=self.feedback_cb)
        self._send_goal_future.add_done_callback(self.goal_response_cb)

    def goal_response_cb(self, future):
        goal_handle = future.result()
        if not goal_handle.accepted:
            self.get_logger().error('Goal rejected!')
            return
        self.get_logger().info('Goal accepted!')
        self._get_result_future = goal_handle.get_result_async()
        self._get_result_future.add_done_callback(self.get_result_cb)

    def feedback_cb(self, feedback_msg):
        seq = feedback_msg.feedback.partial_sequence
        self.get_logger().info(f'Feedback: {seq}')

    def get_result_cb(self, future):
        result = future.result().result
        self.get_logger().info(f'Result: {result.sequence}')
        rclpy.shutdown()

def main(args=None):
    rclpy.init(args=args)
    client = FibonacciClient()
    client.send_goal(10)
    rclpy.spin(client)

from my_interfaces.msg import Robot
from my_interfaces.srv import SetVelocity
from my_interfaces.action import Navigate

class MyRobotNode(Node):
    def __init__(self):
        super().__init__('my_robot_node')

        # Publisher using custom msg
        self.pub = self.create_publisher(Robot, '/robot_status', 10)

        # Service server using custom srv
        self.srv = self.create_service(
            SetVelocity, 'set_velocity', self.set_velocity_cb)

        self.timer = self.create_timer(1.0, self.publish_status)

    def publish_status(self):
        msg = Robot()
        msg.robot_name = 'R2D2'
        msg.robot_id = 1
        msg.battery_level = 87.5
        msg.is_active = True
        msg.joint_angles = [0.0, 0.5, 1.2, -0.3, 0.8, 0.0]
        self.pub.publish(msg)

    def set_velocity_cb(self, request, response):
        self.get_logger().info(
            f'Setting velocity: linear={request.linear_velocity}')
        response.success = True
        response.message = 'Velocity set successfully'
        response.actual_velocity = request.linear_velocity
        return response

Parameters
Config
ℹ️ Parameters
Node configuration values that can be set at startup or changed at runtime. Must be declared before use. Supports: bool, int, float, string, and arrays of each.
Declaring & Using Parameters (Python)
python
param_node.py
import rclpy
from rclpy.node import Node
from rcl_interfaces.msg import ParameterDescriptor

class ParamNode(Node):
    def __init__(self):
        super().__init__('param_node')

        # Basic declare with default value
        self.declare_parameter('robot_name', 'my_robot')
        self.declare_parameter('max_speed', 1.5)
        self.declare_parameter('debug_mode', False)

        debug     = self.get_parameter('debug_mode').get_parameter_value().bool_value
        wpts      = self.get_parameter('waypoints').get_parameter_value().double_array_value

        name2 = self.get_parameter('robot_name').value   # same as above

        # Register a callback for live parameter updates
        self.add_on_set_parameters_callback(self.param_callback)

    def param_callback(self, params):
        from rcl_interfaces.msg import SetParametersResult
        for p in params:
            self.get_logger().info(f'Parameter changed: {p.name} = {p.value}')
            if p.name == 'max_speed' and p.value < 0:
                return SetParametersResult(successful=False, reason='Speed must be positive')
        return SetParametersResult(successful=True)

TF2 — Transform System
Advanced
ℹ️ TF2
TF2 tracks coordinate frames and their relationships over time. Every robot part (base, arm, sensor) has a frame. TF2 lets you transform positions between any frames automatically.
Static Transform Broadcaster
python
import rclpy
from rclpy.node import Node
from tf2_ros import StaticTransformBroadcaster
from geometry_msgs.msg import TransformStamped
import math

class StaticBroadcaster(Node):
    def __init__(self):
        super().__init__('static_broadcaster')
        self.tf_static = StaticTransformBroadcaster(self)
        self.send_transform()

    def send_transform(self):
        t = TransformStamped()
        t.header.stamp = self.get_clock().now().to_msg()
        t.header.frame_id = 'base_link'      # parent frame
        t.child_frame_id  = 'camera_link'    # child frame

        # Translation (meters)
        t.transform.translation.x = 0.1
        t.transform.translation.y = 0.0
        t.transform.translation.z = 0.5

        # Rotation as quaternion
        t.transform.rotation.x = 0.0
        t.transform.rotation.y = 0.0
        t.transform.rotation.z = 0.0
        t.transform.rotation.w = 1.0        # no rotation

        self.tf_static.sendTransform(t)

from tf2_ros import TransformBroadcaster
from tf_transformations import quaternion_from_euler  # pip install transforms3d

class DynamicBroadcaster(Node):
    def __init__(self):
        super().__init__('dynamic_broadcaster')
        self.tf = TransformBroadcaster(self)
        self.timer = self.create_timer(0.05, self.broadcast)  # 20Hz
        self.angle = 0.0

    def broadcast(self):
        t = TransformStamped()
        t.header.stamp = self.get_clock().now().to_msg()
        t.header.frame_id = 'world'
        t.child_frame_id  = 'robot'

        t.transform.translation.x = 1.0
        t.transform.translation.y = 0.0
        t.transform.translation.z = 0.0

        # Convert roll, pitch, yaw → quaternion
        q = quaternion_from_euler(0, 0, self.angle)
        t.transform.rotation.x = q[0]
        t.transform.rotation.y = q[1]
        t.transform.rotation.z = q[2]
        t.transform.rotation.w = q[3]

        self.tf.sendTransform(t)
        self.angle += 0.01
Transform Listener (Look Up Transforms)
python
from tf2_ros import Buffer, TransformListener, TransformException
from rclpy.duration import Duration

class TFListener(Node):
    def __init__(self):
        super().__init__('tf_listener')
        self.tf_buffer = Buffer()
        self.tf_listener = TransformListener(self.tf_buffer, self)
        self.timer = self.create_timer(0.1, self.lookup)

    def lookup(self):
        try:
            # Look up transform: where is 'target_frame' relative to 'source_frame'?
            transform = self.tf_buffer.lookup_transform(
                'base_link',          # target frame
                'camera_link',        # source frame
                rclpy.time.Time(),    # latest available
                timeout=Duration(seconds=1.0))

            tx = transform.transform.translation.x
            ty = transform.transform.translation.y
            tz = transform.transform.translation.z
            self.get_logger().info(f'Camera at: ({tx:.2f}, {ty:.2f}, {tz:.2f})')

        except TransformException as e:
            self.get_logger().warn(f'Transform error: {e}')

Executors & Callback Groups
Advanced
⚠️ By Default
rclpy.spin(node) uses a SingleThreadedExecutor. Callbacks are queued and run one at a time. A slow callback will block all others. Use MultiThreadedExecutor + callback groups to run callbacks concurrently.
MultiThreadedExecutor with Callback Groups
python
import rclpy
from rclpy.node import Node
from rclpy.executors import MultiThreadedExecutor, SingleThreadedExecutor
from rclpy.callback_groups import ReentrantCallbackGroup, MutuallyExclusiveCallbackGroup

class MyNode(Node):
    def __init__(self):
        super().__init__('my_node')

        # ReentrantCallbackGroup: callbacks in this group CAN run concurrently
        self.reentrant_group = ReentrantCallbackGroup()

        # MutuallyExclusiveCallbackGroup: only ONE callback at a time
        # (but can run concurrently with other groups)
        self.exclusive_group = MutuallyExclusiveCallbackGroup()

        # Assign callbacks to groups
        self.timer1 = self.create_timer(
            0.1, self.fast_callback,
            callback_group=self.reentrant_group)

        self.sub = self.create_subscription(
            String, '/topic', self.slow_callback,
            10, callback_group=self.exclusive_group)

        self.srv = self.create_service(
            AddTwoInts, 'add',
            self.service_callback,
            callback_group=self.exclusive_group)

    def fast_callback(self):
        pass  # runs at 10Hz without blocking others

    def slow_callback(self, msg):
        import time
        time.sleep(0.5)  # With MultiThreaded, doesn't block fast_callback

    def service_callback(self, req, res):
        res.sum = req.a + req.b
        return res

def main(args=None):
    rclpy.init(args=args)
    node = MyNode()

    # Use multiple threads — pass num_threads or leave for auto
    executor = MultiThreadedExecutor(num_threads=4)
    executor.add_node(node)

    try:
        executor.spin()
    finally:
        executor.shutdown()
        node.destroy_node()
        rclpy.shutdown()

class AsyncClientNode(Node):

Creating a Lifecycle Node (Python)
python
lifecycle_node.py
import rclpy
from rclpy.lifecycle import LifecycleNode, TransitionCallbackReturn, State

class MyLifecycleNode(LifecycleNode):
    def __init__(self):
        super().__init__('my_lifecycle_node')
        self.pub_ = None
        self.timer_ = None

    # Called when: configure transition is triggered
    def on_configure(self, state: State) -> TransitionCallbackReturn:
        self.get_logger().info('Configuring...')
        # Setup publishers, subscribers, parameters here
        # Do NOT start timers or start active work here
        self.pub_ = self.create_lifecycle_publisher(String, '/data', 10)
        return TransitionCallbackReturn.SUCCESS  # or FAILURE / ERROR

    # Called when: activate transition is triggered
    def on_activate(self, state: State) -> TransitionCallbackReturn:
        self.get_logger().info('Activating...')
        # Start publishing, begin active work
        self.timer_ = self.create_timer(1.0, self.publish_data)
        return super().on_activate(state)  # activates the publisher

    # Called when: deactivate transition is triggered
    def on_deactivate(self, state: State) -> TransitionCallbackReturn:
        self.get_logger().info('Deactivating...')
        self.timer_.cancel()
        return super().on_deactivate(state)  # deactivates publisher

    # Called when: cleanup transition is triggered
    def on_cleanup(self, state: State) -> TransitionCallbackReturn:
        self.get_logger().info('Cleaning up...')
        self.destroy_publisher(self.pub_)
        self.pub_ = None
        return TransitionCallbackReturn.SUCCESS

    # Called when: shutdown transition is triggered
    def on_shutdown(self, state: State) -> TransitionCallbackReturn:
        self.get_logger().info('Shutting down...')
        return TransitionCallbackReturn.SUCCESS

    def publish_data(self):
        if self.pub_ is not None and self.pub_.is_activated:
            msg = String()
            msg.data = 'alive'
            self.pub_.publish(msg)

def main(args=None):
    rclpy.init(args=args)
    node = MyLifecycleNode()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()

#!/usr/bin/env python3
"""Template: copy this as starting point for any new node."""
import rclpy
from rclpy.node import Node
from std_msgs.msg import String

class TemplateNode(Node):
    def __init__(self):
        super().__init__('template_node')

        # Parameters
        self.declare_parameter('my_param', 'default_value')
        self.my_param = self.get_parameter('my_param').value

        # Publisher
        self.publisher = self.create_publisher(String, '/output', 10)

        # Subscriber
        self.subscription = self.create_subscription(
            String, '/input', self.input_callback, 10)

        # Timer
        self.timer = self.create_timer(1.0, self.timer_callback)

        self.get_logger().info(f'Node started with param: {self.my_param}')

    def timer_callback(self):
        msg = String()
        msg.data = 'Hello from template!'
        self.publisher.publish(msg)

    def input_callback(self, msg: String):
        self.get_logger().info(f'Received: {msg.data}')

def main(args=None):
    rclpy.init(args=args)
    node = TemplateNode()
    try:
        rclpy.spin(node)
    except KeyboardInterrupt:
        pass
    finally:
        node.destroy_node()
        rclpy.shutdown()

if __name__ == '__main__':
    main()
